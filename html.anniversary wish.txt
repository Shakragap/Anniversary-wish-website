  <script>
    // --- Audio Synthesis Setup ---
    let soundEnabled = true;
    let audioCtx = null;

    function getAudioContext() {
      if (!audioCtx) {
        audioCtx = new (window.AudioContext || window.webkitAudioContext)();
      }
      if (audioCtx.state === 'suspended') {
        audioCtx.resume();
      }
      return audioCtx;
    }

    function playChime(frequency = 523.25, duration = 1.5, type = 'sine') {
      if (!soundEnabled) return;
      try {
        const ctx = getAudioContext();
        const now = ctx.currentTime;
        const osc = ctx.createOscillator();
        const osc2 = ctx.createOscillator();
        const gainNode = ctx.createGain();
        const lowpass = ctx.createBiquadFilter();

        osc.type = type;
        osc.frequency.setValueAtTime(frequency, now);

        osc2.type = 'triangle';
        osc2.frequency.setValueAtTime(frequency * 2, now);

        lowpass.type = 'lowpass';
        lowpass.frequency.setValueAtTime(1500, now);
        lowpass.frequency.exponentialRampToValueAtTime(300, now + duration);

        gainNode.gain.setValueAtTime(0, now);
        gainNode.gain.linearRampToValueAtTime(0.15, now + 0.05);
        gainNode.gain.exponentialRampToValueAtTime(0.0001, now + duration);

        osc.connect(lowpass);
        osc2.connect(lowpass);
        lowpass.connect(gainNode);
        gainNode.connect(ctx.destination);

        osc.start(now);
        osc2.start(now);
        osc.stop(now + duration);
        osc2.stop(now + duration);
      } catch (e) {
        console.warn('Audio synthesis failed:', e);
      }
    }

    function playProgression() {
      const notes = [440.00, 554.37, 659.25, 880.00];
      notes.forEach((freq, idx) => {
        setTimeout(() => playChime(freq, 2.0), idx * 120);
      });
    }

    function playCelebrationBells() {
      const notes = [261.63, 293.66, 329.63, 392.00, 440.00, 523.25, 587.33, 659.25, 783.99, 880.00, 1046.50];
      notes.forEach((freq, idx) => {
        setTimeout(() => {
          playChime(freq, 2.5, idx % 2 === 0 ? 'sine' : 'triangle');
        }, idx * 80);
      });
    }

    // Toggle Sound Button handler

    soundToggle.addEventListener('click', () => {

        playChime(523.25, 1.5);
      } else {

    // --- Three.js 3D Floating Hearts Engine ---
    const heartsContainer = document.getElementById('hearts-container');

    renderer.setSize(width, height);
    renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
    heartsContainer.appendChild(renderer.domElement);

    // Dynamic Ambient/Point/Directional lighting
    const ambientLight = new THREE.AmbientLight('#ffffff', 0.6);

    pinkDirectional.position.set(-5, 8, 4);

    // Heart geometry shape construction

    heartShape.moveTo(0, 0.4);

    const geometry = new THREE.ExtrudeGeometry(heartShape, extrudeSettings);
    geometry.center();

    const colors = ['#ff0a54', '#ff477e', '#ff7096', '#ff85a1', '#ff5c8a', '#c9184a'];
    const hearts = [];
    const heartsCount = 24;
    let isCelebrating = false;

    for (let i = 0; i < heartsCount; i++) {
      const color = colors[i % colors.length];
      const material = new THREE.MeshStandardMaterial({
        color: new THREE.Color(color),
        roughness: 0.25,
        metalness: 0.2,
        transparent: true,
        opacity: 0.85,
        side: THREE.DoubleSide
      });

      const mesh = new THREE.Mesh(geometry, material);
      const x = (Math.random() - 0.5) * 16;
      const y = (Math.random() - 0.5) * 14;
      const z = (Math.random() - 0.5) * 6 - 2;
      mesh.position.set(x, y, z);

      const scale = 0.5 + Math.random() * 0.7;
      mesh.scale.set(scale, scale, scale);

      mesh.rotation.set(
        Math.random() * Math.PI,
        Math.random() * Math.PI,
        Math.random() * Math.PI
      );

      scene.add(mesh);

      hearts.push({
        mesh,
        material,
        speedY: 0.012 + Math.random() * 0.012,
        speedRotX: (Math.random() - 0.5) * 0.015,
        speedRotY: (Math.random() - 0.5) * 0.015,
        speedRotZ: (Math.random() - 0.5) * 0.015,
        sinOffset: Math.random() * 100,
        sinSpeed: 0.5 + Math.random() * 1.5,
        initialX: x,
        initialZ: z
      });
    }

    // Floating animation loop
    const clock = new THREE.Clock();
    function animate() {
      requestAnimationFrame(animate);
      const elapsedTime = clock.getElapsedTime();

      hearts.forEach((heart) => {
        heart.mesh.rotation.x += heart.speedRotX;
        heart.mesh.rotation.y += heart.speedRotY;
        heart.mesh.rotation.z += heart.speedRotZ;

        if (isCelebrating) {
          if (heart.swirlVelocityX === undefined) {
            const dir = Math.sign(heart.mesh.position.x) || (Math.random() > 0.5 ? 1 : -1);
            heart.swirlVelocityX = dir * (0.05 + Math.random() * 0.08);
          }
          heart.mesh.position.x += heart.swirlVelocityX;
          heart.mesh.position.y += 0.08 + heart.speedY * 2;
          heart.mesh.rotation.y += 0.04;

          if (heart.material.opacity > 0) {
            heart.material.opacity -= 0.005;
          }
        } else {
          heart.mesh.position.y += heart.speedY;
          heart.mesh.position.x += Math.sin(elapsedTime * heart.sinSpeed + heart.sinOffset) * 0.005;

          if (heart.mesh.position.y > 7) {
            heart.mesh.position.y = -7;
            heart.mesh.position.x = (Math.random() - 0.5) * 16;
            heart.material.opacity = 0.85;
            heart.swirlVelocityX = undefined;
          }
        }
      });

      renderer.render(scene, camera);
    }
    animate();

    window.addEventListener('resize', () => {
      const w = window.innerWidth;
      const h = window.innerHeight;
      camera.aspect = w / h;
      camera.updateProjectionMatrix();
      renderer.setSize(w, h);
    });


    // --- Step Navigation & GSAP Core Transitions ---
    let currentStep = 1;

    function goToStep(nextStep) {
      if (soundEnabled) {
        if (nextStep === 5) {
          playProgression();
        } else {
          playChime(440 + nextStep * 110, 1.2);
        }
      }

      const currentEl = document.getElementById(`step-${currentStep}`);
      const nextEl = document.getElementById(`step-${nextStep}`);

      // Smooth exit transition for current card
      gsap.to(currentEl, {
        opacity: 0,
        y: -30,
        scale: 0.95,
        duration: 0.5,
        ease: 'power3.inOut',
        onComplete: () => {
          currentEl.classList.add('hidden');
          nextEl.classList.remove('hidden');

          // Smooth enter transition for next card
          gsap.fromTo(nextEl, 
            { opacity: 0, y: 30, scale: 0.95 },
            { opacity: 1, y: 0, scale: 1, duration: 0.6, ease: 'power3.out' }
          );

          // Stagger child visual elements within next step
          const staggeredElements = nextEl.querySelectorAll('h2, p, button, .grid, .polaroid-wrapper');
          if (staggeredElements.length > 0) {
            gsap.fromTo(staggeredElements,
              { opacity: 0, y: 15 },
              { opacity: 1, y: 0, duration: 0.5, stagger: 0.1, ease: 'power2.out' }
            );
          }
        }
      });

      currentStep = nextStep;

      // Fill up horizontal progress bar seamlessly
      const progressPercent = (nextStep / 5) * 100;
      document.getElementById('progress-bar').style.width = `${progressPercent}%`;
    }


    // --- Interactive Parallax Polaroid ---
    const polaroidBox = document.getElementById('polaroid-box');
    if (polaroidBox) {
      polaroidBox.addEventListener('mousemove', (e) => {
        const rect = polaroidBox.getBoundingClientRect();
        const x = (e.clientX - rect.left) / rect.width - 0.5;
        const y = (e.clientY - rect.top) / rect.height - 0.5;
        
        gsap.to(polaroidBox, {
          rotateX: -y * 24,
          rotateY: x * 24,
          scale: 1.05,
          duration: 0.2,
          ease: 'power1.out'
        });
      });

      polaroidBox.addEventListener('mouseleave', () => {
        gsap.to(polaroidBox, {
          rotateX: -1,
          rotateY: 3,
          scale: 1,
          duration: 0.5,
          ease: 'power2.out'
        });
      });
    }


    // --- The Grand Finale Sequence ---
    function triggerGrandFinale() {
      isCelebrating = true;
      if (soundEnabled) {
        playCelebrationBells();
      }

      // Disable and fade celebrate button out
      const btnCelebrate = document.getElementById('btn-celebrate');
      gsap.to(btnCelebrate, {
        opacity: 0,
        scale: 0.8,
        y: 15,
        duration: 0.4,
        onComplete: () => {
          btnCelebrate.remove();
        }
      });

      // Powerful dual-corner confetti explosions
      confetti({
        particleCount: 100,
        angle: 60,
        spread: 80,
        origin: { x: 0, y: 0.8 },
        colors: ['#ff0a54', '#ff477e', '#ff7096', '#ff85a1', '#ffe5ec', '#ffb703'],
      });
      confetti({
        particleCount: 100,
        angle: 120,
        spread: 80,
        origin: { x: 1, y: 0.8 },
        colors: ['#ff0a54', '#ff477e', '#ff7096', '#ff85a1', '#ffe5ec', '#ffb703'],
      });

      // Continuous raining cascade of confetti & emojis
      const duration = 5000;
      const end = Date.now() + duration;

      const interval = setInterval(() => {
        const timeLeft = end - Date.now();
        if (timeLeft <= 0) {
          return clearInterval(interval);
        }

        // Circular standard confetti
        confetti({
          particleCount: 15,
          startVelocity: 0,
          ticks: 150,
          origin: { x: Math.random(), y: Math.random() * 0.2 - 0.05 },
          colors: ['#ff0a54', '#ff7096', '#ffccd5', '#ffb703', '#ffffff'],
          gravity: 0.75,
          scalar: 1.1,
          drift: (Math.random() - 0.5) * 1.2,
        });

        // Romantic emoji confetti showers
        confetti({
          particleCount: 6,
          startVelocity: 0,
          ticks: 150,
          origin: { x: Math.random(), y: Math.random() * 0.2 - 0.05 },
          shapes: ['text'],
          shapeOptions: {
            text: {
              value: ['❤️', '💖', '✨', '🌹']
            }
          },
          gravity: 0.65,
          scalar: 2.2,
          drift: (Math.random() - 0.5) * 1.5,
        });
      }, 250);

      // Elegantly fade and slide final wish into view
      const finalMessageBox = document.getElementById('final-message-box');
      finalMessageBox.classList.remove('hidden');
      gsap.to(finalMessageBox, {
        opacity: 1,
        y: 0,
        scale: 1,
        duration: 0.8,
        delay: 0.8,
        ease: 'back.out(1.7)'
      });
    }
